import os
import urllib.parse
import pdfkit

# 1. Update the function signature to accept the config
def download_urls_as_pdf(url_file_path, output_directory, config):
    """
    Reads a file containing URLs and saves each as a PDF.
    """
    if not os.path.exists(output_directory):
        os.makedirs(output_directory)

    try:
        with open(url_file_path, 'r', encoding='utf-8') as file:
            urls = [line.strip() for line in file if line.strip()]
    except FileNotFoundError:
        print(f"Error: The file '{url_file_path}' was not found.")
        return

    index = 0
    for url in urls:
        index += 1
        try:
            
            parsed_url = urllib.parse.urlparse(url)
            safe_filename = parsed_url.netloc + parsed_url.path.replace('/', '_')
            safe_filename = safe_filename.strip('_')
            safe_filename += str(index)
            if not safe_filename.endswith('.pdf'):
                safe_filename += '.pdf'
            
            output_path = os.path.join(output_directory, safe_filename)

            print(f"Converting: {url}")
            
            # 2. Pass the config object here!
            pdfkit.from_url(url, output_path, configuration=config)
            
            print(f"Success -> Saved as {safe_filename}\n")

        except Exception as e:
            print(f"Failed to process {url}.")
            print(f"Error details: {e}\n")

if __name__ == "__main__":
    # 3. Capture the configuration in a variable
    path_to_wkhtmltopdf = r'C:\Program Files\wkhtmltopdf\bin\wkhtmltopdf.exe'
    config = pdfkit.configuration(wkhtmltopdf=path_to_wkhtmltopdf)
    
    SCRIPT_DIR = os.path.dirname(os.path.abspath(__file__))
    PROJECT_ROOT = os.path.dirname(SCRIPT_DIR)

    INPUT_FILE = os.path.join(PROJECT_ROOT, "notes", "medicare-urls.txt") 
    OUTPUT_FOLDER = os.path.join(PROJECT_ROOT, "policies", "medicare") 
    
    # 4. Pass the config into your helper function
    download_urls_as_pdf(INPUT_FILE, OUTPUT_FOLDER, config)
